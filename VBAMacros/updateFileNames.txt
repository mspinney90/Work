' Rename all file names in folder based on array in excel.
' Update pathing. Remove file extension, but add it back in after file name added.
' Keep column A of original file names in column A.
'**Make sure \ after file name in file pathing. Function won't rename without (macro for getFileNames() does not need it, but part 2 does.

Sub updateFileNames()
  Dim myPath AS String
  myPath = "C:\"

  r = 2
  Do Until IsEmpty(Cells(r,1)) And IsEmpty(Cells(r,2))
    Name myPath & Cells(r,1).Value As myPath & Cells(r,2).Value
    r = r + 1
  Loop

End Sub
